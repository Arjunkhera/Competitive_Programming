vector<int> answer;
if(!root) return answer;

TreeNode *temp,*prev = nullptr,*cur = root;
stack<TreeNode*> stk;
while(cur || !stk.empty())
  if(cur){ stk.push(cur); cur = cur->left; }
  else{
    temp = stk.top();
    if(!temp->right || temp->right == prev){ prev = temp; answer.push_back(temp->val); stk.pop(); }
    else cur = temp->right;
  }
return answer;
